// why? 
// current code: 
catch(err){res.send(err.message)} or throw new Error("user already exists")

# it just send message 
# And hard to maintain


//
industry solution: custom error class
# that is contains status code, error message, and status
# and one central class, handles all error

src/
├── errors/
│   └── AppError.js


after building this custom Error handler class

# we can send message and status:
 throw new AppError(message, status)


400 = Bad Request  => the user sent data invalid (wrong email without domain, wrong password or incomplete)
401 = Unauthorized => user is not logged in or token invalid
403 = Forbidden => user logged in but doesn't have permission (only admin can delete)
404 = Not Found => requested resource doesn't exist (user/product not found) 
409 = Conflict => data already exists or conflict with current data (email already exists/ username already taken)
500 = Internal Server Error => something broke on the Server (db crash) — we don't handle it manually, it handle global middleware
